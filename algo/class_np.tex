\section{Complexity class NP}
\begin{definition}
	\textit{Class NP} is the class of problems 
\end{definition}

\begin{theorem}
	$P \subset NP$
\end{theorem}

\begin{theorem}
	Let $\mathcal{L} \in NP$ there exist a polynomial p(n) and TM
	for solving $\mathcal{L}$ with complexity $O(2^{p(n)})$.
\end{theorem}
